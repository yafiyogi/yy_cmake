# From https://best.openssf.org/Compiler-Hardening-Guides/Compiler-Options-Hardening-Guide-for-C-and-C++.html
add_compile_options(
  -pedantic
  -Wall
  -Werror
  -Wextra
  -Wformat
  -Wformat=2
  -Wconversion
  -Wimplicit-fallthrough
  -Wnon-virtual-dtor
  -Wnull-dereference
  -Woverloaded-virtual
  -Wsign-conversion
  -Wshadow
  -Wunused
  -U_FORTIFY_SOURCE
  -D_FORTIFY_SOURCE=3
  -fstack-protector-strong
  -fcf-protection=full
  -fPIE
  -fno-delete-null-pointer-checks
  -fno-strict-overflow
  -fno-strict-aliasing
)

string(TOLOWER "${CMAKE_CXX_COMPILER_ID}" yy_cxx_compile_id)
string(TOLOWER "${CMAKE_BUILD_TYPE}" yy_build_type)
string(TOLOWER "${CMAKE_SYSTEM_NAME}" yy_system_name) # see https://cmake.org/cmake/help/latest/variable/CMAKE_SYSTEM_NAME.html

if (yy_cxx_compile_id STREQUAL "clang")
  # using Clang
  include(${YY_CMAKE}/cmake_llvm.txt)
elseif (yy_cxx_compile_id STREQUAL "gnu")
  # using GCC
  include(${YY_CMAKE}/cmake_gcc.txt)
endif()
