SET(YY_DEBUG_FLAGS "-g")

# Linux
SET(YY_LINUX_GCC_OPTIONS
    "" )

# MacOs
SET(YY_MACOS_GCC_OPTIONS
    "" )

if("${yy_system_name}" STREQUAL "darwin")
  set(YY_GCC_OPTIONS "${YY_MACOS_GCC_OPTIONS}")
elseif("${yy_system_name}" STREQUAL "linux")
  set(YY_GCC_OPTIONS "${YY_LINUX_GCC_OPTIONS}")
else()
  set(YY_GCC_OPTIONS "")
endif()

# From https://best.openssf.org/Compiler-Hardening-Guides/Compiler-Options-Hardening-Guide-for-C-and-C++.html
add_compile_options(
  -Wduplicated-cond
  -Wduplicated-branches
  -Wlogical-op
  -Wtrampolines
  -Wtrivial-auto-var-init
  -Wuseless-cast
  # 2024-05-22 -D_GLIBCXX_ASSERTIONS
  # g++12    -fstrict-flex-arrays=3
  # 2024-05-22 -fstack-clash-protection
  # -mbranch-protection=standard

  # optimizations
  -finline-functions
  --param max-inline-insns-auto=100
  #--param early-inlining-insns=200

  "${YY_GCC_OPTIONS}"

)
